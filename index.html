    TÍNH VÀ TIN RA TỔNG CÁC SỐ CHIA HẾT CHO SỐ NGUYÊN K

    Var A:array[0..250] of integer;
    i,N:byte;
    k:integer;
    tong:longint;
Begin
    writeln('nhap so phan tu');
    readln(N);
    for i:=1 to N do
    begin
        writeln('nhap phan tu so ',i);
        readln(A[i]);
    end;
    writeln('nhap so k ');
    readln(k);
    tong:=0;
    for i:=1 to N do
        if (A[i] mod k = 0) then
            tong:=tong + A[i];
    writeln('tong la ',tong);
    readln;
End.

    mod dùng làm gì

    > chia lấy phần dư, xem có chia hết không

    ĐẾM SỐ KÍ TỰ h CÓ TRONG XÂU

    Var str:string;
    i, dem, len:integer;
Begin
    writeln('Nhap xau ');
    readln(str);
    len:=length(str);
    dem:=0;
    for i:=1 to len do
        if str[i] = 'h' then
            dem:=dem+1;
    writeln('So ki tu h trong xau la ',dem);
    readln;
End.
