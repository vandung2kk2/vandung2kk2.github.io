TÍNH VÀ TIN RA TỔNG CÁC SỐ CHIA HẾT CHO SỐ NGUYÊN K

Var A:array[0..250] of integer;
    i,N:byte;
    k:integer;
    tong:longint;
Begin
    writeln('nhap so phan tu');
    readln(N);
    for i:=1 to N do
    begin
        writeln('nhap phan tu so ',i);
        readln(A[i]);
    end;
    writeln('nhap so k ');
    readln(k);
    tong:=0;
    for i:=1 to N do
        if (A[i] mod k = 0) then
            tong:=tong + A[i];
    writeln('tong la ',tong);
    readln;
End.
